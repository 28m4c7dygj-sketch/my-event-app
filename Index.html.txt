        function adjustLayout(category, type, delta) {
            const activeEvent = getActiveEvent();
            if (!activeEvent) return;

            const config = activeEvent.seatingConfig[category];
            
            if (type === 'rows') {
                const currentCount = config.rows.length;
                const newCount = currentCount + delta;
                
                if (newCount < 1) {
                    showToast('يجب أن يحتوي المدرج على صف واحد على الأقل!', 'error');
                    return;
                }
                if (newCount > 15) {
                    showToast('الحد الأقصى المسموح به هو 15 صفاً لسلامة العرض!', 'error');
                    return;
                }

                if (category === 'VVIP') {
                    config.rows = generateAlphabetRows(newCount, 'A');
                    const lastVvipCharCode = config.rows[config.rows.length - 1].charCodeAt(0);
                    const startVipChar = String.fromCharCode(lastVvipCharCode + 1);
                    activeEvent.seatingConfig.VIP.rows = generateAlphabetRows(activeEvent.seatingConfig.VIP.rows.length, startVipChar);
                } else {
                    const vvipLastChar = activeEvent.seatingConfig.VVIP.rows[activeEvent.seatingConfig.VVIP.rows.length - 1];
                    const startVipChar = String.fromCharCode(vvipLastChar.charCodeAt(0) + 1);
                    config.rows = generateAlphabetRows(newCount, startVipChar);
                }

            } else if (type === 'seats') {
                const newSeats = config.seatsPerRow + delta;
                if (newSeats < 5) {
                    showToast('يجب أن يحتوي الصف على 5 مقاعد على الأقل لجمالية التنظيم!', 'error');
                    return;
                }
                if (newSeats > 15) {
                    showToast('الحد الأقصى هو 15 مقعداً لكل صف لمنع ضيق المنصة!', 'error');
                    return;
                }
                config.seatsPerRow = newSeats;
            }

            sanitizeGuestsAfterLayoutChange();
            saveEventsToLocalStorage();
            updateConfigUI();
            adaptSeatOptions();
            renderVisualPlatform();
            renderTable();
            updateStats();
        }

        function sanitizeGuestsAfterLayoutChange() {
            const activeEvent = getActiveEvent();
            if (!activeEvent) return;

            let affectedGuests = [];

            activeEvent.guests.forEach(g => {
                if (g.approval === 'approved' && g.row && g.seatNum) {
                    const config = activeEvent.seatingConfig[g.category];
                    const isRowValid = config.rows.includes(g.row);
                    const isSeatValid = parseInt(g.seatNum) <= config.seatsPerRow;

                    if (!isRowValid || !isSeatValid) {
                        g.row = null; // إزالة تعيين المقعد
                        g.seatNum = null;
                        affectedGuests.push(g.name);
                    }
                }
            });

            if (affectedGuests.length > 0) {
                setTimeout(() => {
                    const formattedNames = affectedGuests.slice(0, 3).join('، ') + (affectedGuests.length > 3 ? ' وثلة آخرين' : '');

        function adaptProtocolSeats() {
            const select = document.getElementById('prot-guest-select');
            const rowSelect = document.getElementById('prot-row-select');
            const seatSelect = document.getElementById('prot-seat-select');
            if (!select || !rowSelect || !seatSelect) return;

            if (!activeEvent || !guestId || !row) return;

            if (!guest) return;

            seatSelect.innerHTML = '';
            const config = activeEvent.seatingConfig[guest.category];

            // البحث عن الكراسي غير المحجوزة في هذا الصف

        // معالجة تسكين وتوزيع الكراسي من قبل المراسم
        function handleProtocolSeating(e) {
            e.preventDefault();
            const activeEvent = getActiveEvent();
            if (!activeEvent) return;

            const guestId = document.getElementById('prot-guest-select').value;
            const row = document.getElementById('prot-row-select').value;
            const seatNum = parseInt(document.getElementById('prot-seat-select').value);

            if (!guestId || !row || !seatNum) {
                showToast('تنبيه المراسم: يرجى اختيار ضيف وتحديد صف ومقعد شاغر للتسكين!', 'error');
                return;
            }

            const guest = activeEvent.guests.find(g => g.id == guestId);
            if (!guest) return;

            if (isSeatTaken(guest.category, row, seatNum)) {
                showToast('خطأ المراسم: هذا المقعد محجوز لضيف آخر!', 'error');
                return;
            }

            // تخصيص الكرسي للضيف
            guest.row = row;
            guest.seatNum = seatNum;

            showToast(`تم تسكين الضيف (${guest.name}) في المقعد ${row}-${seatNum} بنجاح`, 'success');
            
            saveEventsToLocalStorage();
            syncActiveEventData();
        }

        // معالجة إرسال العلاقات العامة للدعوات وحفظ البيانات
        function handleFormSubmit(e) {
            e.preventDefault();

            const activeEvent = getActiveEvent();
            if (!activeEvent) return;

            const editId = document.getElementById('edit-guest-id').value;
            const name = document.getElementById('g-input-name').value.trim();
            const job = document.getElementById('g-input-job').value.trim();
            const phone = document.getElementById('g-input-phone').value.trim();
            const email = document.getElementById('g-input-email').value.trim() || 'لا يوجد بريد';
            const tickets = parseInt(document.getElementById('g-input-tickets').value) || 1;
            const category = document.getElementById('g-input-category').value;
            const approval = document.getElementById('g-input-approval').value;

            if (editId) {
                const index = activeEvent.guests.findIndex(g => g.id === parseFloat(editId));
                if (index !== -1) {
                    const oldApproval = activeEvent.guests[index].approval;
                    
                    // إذا تحولت حالة الحضور لغير موافق، نزيل الكرسي تلقائياً
                    let clearSeat = false;
                    if (approval !== 'approved' && oldApproval === 'approved') {
                        clearSeat = true;
                    }

                    activeEvent.guests[index] = {
                        ...activeEvent.guests[index],
                        name, job, phone, email, tickets, category, approval,
                        row: clearSeat ? null : activeEvent.guests[index].row,
                        seatNum: clearSeat ? null : activeEvent.guests[index].seatNum
                    };
                    showToast('تم تعديل بيانات الضيف بنجاح وحفظ السجلات للجنة', 'success');
                }
            } else {
                // تسجيل ضيف جديد: العلاقات العامة لا تحدد كرسي، يتم حفظه بدون كرسي
                activeEvent.guests.push({
                    id: Date.now(),
                    name, job, phone, email, tickets, category, approval,
                    row: null,
                    seatNum: null
                });
                showToast('تم تسجيل بيانات الضيف الجديد بنجاح في سجلات العلاقات الدولية', 'success');
            }

            saveEventsToLocalStorage();
            syncActiveEventData();
            resetForm();
        }

        // تصفير وإخلاء الكراسي أو تعديل الحالة بشكل دائري
        function toggleApproval(id) {
            const activeEvent = getActiveEvent();
            if (!activeEvent) return;

            const guest = activeEvent.guests.find(g => g.id === id);
            if (!guest) return;

                guest.row = null; // تفريغ المقعد تلقائياً

                guest.approval = 'pending';
                showToast(`تم تعيين الضيف ${guest.name} كـ (معلق الحضور)`, 'success');
            } else {
                guest.approval = 'approved';
                showToast(`تم الموافقة على الضيف ${guest.name}، بانتظار تسكينه من قبل المراسم`, 'success');
            }

            syncActiveEventData();
        }

        function buildSectionGrid(container, category) {

            if (!config) return;

                seatsWrapper.style.gridTemplateColumns = `repeat(${config.seatsPerRow}, minmax(0, 1fr))`;

                for (let s = 1; s <= config.seatsPerRow; s++) {

                        g.approval === 'approved' && 
                        g.category === category && 
                        g.row === rowLetter &&

                    seatBtn.className = 'interactive-seat aspect-square rounded-lg flex flex-col items-center justify-center font-mono font-bold text-[10px] transition-all relative border shadow-sm';

                        seatBtn.onclick = () => showGuestModal(occupant);
                    } else {

        // اختيار الكرسي من الرادار لنموذج المراسم
        function selectSeatInProtocolForm(category, row, seatNum) {
            if (currentUser.role === 'relations') {
                showToast('تنبيه: لا تمتلك صلاحيات تسكين الكراسي، يرجى التنسيق مع قسم المراسم!', 'error');
                return;
            }

            const activeEvent = getActiveEvent();
            if (!activeEvent) return;

            // فتح لوحة التسكين وتحديد الكرسي
            const rowSelect = document.getElementById('prot-row-select');
            const seatSelect = document.getElementById('prot-seat-select');
            
            if (rowSelect && seatSelect) {
                rowSelect.value = row;
                
                const guestSelect = document.getElementById('prot-guest-select');
                const guestId = guestSelect.value;
                const guest = activeEvent.guests.find(g => g.id == guestId);
                
                if (guest) {
                    seatSelect.innerHTML = '';
                    const config = activeEvent.seatingConfig[guest.category];
                    for (let i = 1; i <= config.seatsPerRow; i++) {
                        const isTaken = isSeatTaken(guest.category, row, i);
                        if (!isTaken || i === seatNum) {
                            const opt = document.createElement('option');
                            opt.value = i;
                            opt.innerText = `مقعد ${i}`;
                            if (i === seatNum) opt.selected = true;
                            seatSelect.appendChild(opt);
                        }
                    }
                }
            }

            showToast(`تم تحديد الكرسي المستهدف: الصف ${row} - مقعد ${seatNum} برادار المراسم`, 'success');
        }

        // دوال إنشاء رسالة ومشاركة تفاصيل التذكرة
        function generateShareMessage(guest) {
            const activeEvent = getActiveEvent();
            let approvalText = 'معلق الحضور';
            if (guest.approval === 'approved') approvalText = 'موافقة حضور مؤكدة';
            if (guest.approval === 'declined') approvalText = 'اعتذار عن الحضور';

            const dateObj = new Date(activeEvent.date);
            const formattedDate = dateObj.toLocaleDateString('ar-SA', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });

        function shareOnWhatsApp(guest) {
            const cleanedPhone = guest.phone.replace(/[^0-9]/g, ''); 
            const formattedPhone = cleanedPhone.startsWith('0') ? '966' + cleanedPhone.substring(1) : cleanedPhone; 
            const message = generateShareMessage(guest);
            
            const whatsappUrl = `https://wa.me/${formattedPhone}?text=${message}`;
            window.open(whatsappUrl, '_blank');
            showToast(`تم توليد رابط الدعوة ومشاركته عبر الواتساب للضيف ${guest.name}`, 'success');
        }

        function shareOnEmail(guest) {
            const activeEvent = getActiveEvent();
            if (!guest.email || guest.email === 'لا يوجد بريد' || guest.email.trim() === '') {
                showToast('تنبيه: لا يوجد بريد إلكتروني مسجل لهذا الضيف للتواصل!', 'error');
                return;
            }
            const subject = encodeURIComponent(`دعوة رسمية لحضور فعالية (${activeEvent.name}) - سعادة الأستاذ ${guest.name}`);
            const body = generateShareMessage(guest);
            
            const mailtoUrl = `mailto:${guest.email}?subject=${subject}&body=${body}`;
            window.open(mailtoUrl, '_blank');
            showToast(`جاري فتح عميل البريد الإلكتروني لإرسال الدعوة لـ ${guest.name}`, 'success');
        }

            const toast = document.createElement('div');

        // دالة مخصصة لإنشاء خلفية دائرية الحواف لنصوص الـ Canvas ثلاثية الأبعاد لتعمل بشكل متوافق تماماً
        function drawRoundRectSafe(ctx, x, y, width, height, radius) {
            ctx.beginPath();
            ctx.moveTo(x + radius, y);
            ctx.lineTo(x + width - radius, y);
            ctx.quadraticCurveTo(x + width, y, x + width, y + radius);
            ctx.lineTo(x + width, y + height - radius);
            ctx.quadraticCurveTo(x + width, y + height, x + width - radius, y + height);
            ctx.lineTo(x + radius, y + height);
            ctx.quadraticCurveTo(x, y + height, x, y + height - radius);
            ctx.lineTo(x, y + radius);
            ctx.quadraticCurveTo(x, y, x + radius, y);
            ctx.closePath();
        }

        function init3DScene() {
            const container = document.getElementById('canvas-3d-holder');
            if (!container) return;

            // إذا تم إنشاء المشهد مسبقاً، نقوم بتحديث الكراسي ومطابقة الحجم فقط
            if (scene3D) {
                onWindowResize3D();
                update3DChairs();
                animate3D(); // إستئناف حلقة الرندر عند العودة للتبويب
                return;
            }

            const width = container.clientWidth || window.innerWidth * 0.55;
            const height = container.clientHeight || 550;

            // 1. إنشاء المشهد والإضاءة الدافئة
            scene3D = new THREE.Scene();
            scene3D.background = new THREE.Color(0x0f172a); // كحلي داكن متوافق مع لوحة التحكم
            scene3D.fog = new THREE.FogExp2(0x0f172a, 0.015);

            // 2. الكاميرا والمنظور
            camera3D = new THREE.PerspectiveCamera(45, width / height, 0.1, 1000);
            reset3DCamera();

            // 3. المصيّر الـ 3D (Renderer)
            renderer3D = new THREE.WebGLRenderer({ antialias: true, alpha: false });
            renderer3D.setSize(width, height);
            renderer3D.setPixelRatio(Math.min(window.devicePixelRatio, 2));
            renderer3D.shadowMap.enabled = true;
            renderer3D.shadowMap.type = THREE.PCFSoftShadowMap;

            container.innerHTML = '';
            container.appendChild(renderer3D.domElement);

            // 4. عناصر التحكم بالكاميرا (OrbitControls)
            controls3D = new THREE.OrbitControls(camera3D, renderer3D.domElement);
            controls3D.enableDamping = true;
            controls3D.dampingFactor = 0.05;
            controls3D.maxPolarAngle = Math.PI / 2 - 0.05; // منع الكاميرا من النزول تحت الأرضية
            controls3D.minDistance = 6;
            controls3D.maxDistance = 80;

            // 5. الإضاءة المحيطية والمسرحية
            const ambientLight = new THREE.AmbientLight(0xffffff, 0.55);
            scene3D.add(ambientLight);

            const directLight = new THREE.DirectionalLight(0xfffbeb, 0.85); // إضاءة دافئة مسرحية
            directLight.position.set(20, 45, 20);
            directLight.castShadow = true;
            directLight.shadow.mapSize.width = 1024;
            directLight.shadow.mapSize.height = 1024;
            scene3D.add(directLight);

            // إضاءة خضراء ناعمة منعكسة من اتجاه الملعب
            const pitchLight = new THREE.DirectionalLight(0x10b981, 0.35);
            pitchLight.position.set(0, 10, -45);
            scene3D.add(pitchLight);

            // 6. أرضية المقصورة الفاخرة (محاكاة باركيه خشبي / مخملي)
            const floorGeo = new THREE.PlaneGeometry(65, 65);
            const floorMat = new THREE.MeshStandardMaterial({ 
                color: 0x1e293b, 
                roughness: 0.55,
                metalness: 0.15
            });
            const floor = new THREE.Mesh(floorGeo, floorMat);
            floor.rotation.x = -Math.PI / 2;
            floor.receiveShadow = true;
            scene3D.add(floor);

            // حاجز المقصورة الزجاجي الأمامي الممتد
            const glassGeo = new THREE.BoxGeometry(45, 2.5, 0.15);
            const glassMat = new THREE.MeshPhysicalMaterial({
                color: 0xffffff,
                transparent: true,
                opacity: 0.25,
                roughness: 0.1,
                transmission: 0.85,
                thickness: 1.0
            });
            const glassBarrier = new THREE.Mesh(glassGeo, glassMat);
            glassBarrier.position.set(0, 1.25, -18);
            scene3D.add(glassBarrier);

            // 7. بناء كراسي الـ VVIP المائة (10 صفوف × 10 أعمدة)
            build3DChairs();

            // الاستماع إلى النقرات لاختيار المقاعد عبر الأشعة ثلاثية الأبعاد
            renderer3D.domElement.addEventListener('click', onCanvasClick3D, false);

            // بدء حلقة الرسوم المتحركة والتصيير المستمرة
            animate3D();

            // تصحيح الأبعاد بعد التحميل لضمان دقة الكاميرا
            setTimeout(onWindowResize3D, 150);
        }

        // بناء كراسي الـ VVIP ومصفوفة الإسناد
        function build3DChairs() {
            clickableObjects3D = [];
            chairGroupMap3D.clear();

            // نقوم ببناء كراسي VVIP تفاعلياً لصفوف VVIP الحالية بالفعالية النشطة
            const activeEvent = getActiveEvent();
            if (!activeEvent) return;

            const config = activeEvent.seatingConfig.VVIP;
            const rows = config.rows.length;
            const cols = config.seatsPerRow;

            const xSpacing = 3.5;
            const zSpacing = 3.6;
            
            const xStart = -((cols - 1) * xSpacing) / 2;
            const zStart = -((rows - 1) * zSpacing) / 2 + 4;

            for (let r = 0; r < rows; r++) {
                const rowLetter = config.rows[r];
                for (let c = 0; c < cols; c++) {
                    const seatNum = c + 1;
                    const seatId = `${rowLetter}-${seatNum}`; // المعرف (مثال: A-5)
                    
                    const posX = xStart + (c * xSpacing);
                    const posZ = zStart + (r * zSpacing);
                    const posY = r * 0.16; // تدرج الارتفاع لتأثير الرؤية المدرجة

                    const chairGroup = createVVIPChairMesh(seatId);
                    chairGroup.position.set(posX, posY, posZ);
                    scene3D.add(chairGroup);

                    chairGroupMap3D.set(seatId, {
                        group: chairGroup,
                        row: rowLetter,
                        seatNum: seatNum,
                        posX: posX,
                        posY: posY,
                        posZ: posZ
                    });
                }
            }

            update3DChairs();
        }

        // بناء وتصميم هندسة الكرسي الفردي
        function createVVIPChairMesh(seatId) {
            const group = new THREE.Group();
            group.name = seatId;

            // قاعدة الكرسي ومسند الظهر واليدين
            const baseGeo = new THREE.BoxGeometry(1.7, 0.38, 1.5);
            const baseMat = new THREE.MeshStandardMaterial({ roughness: 0.55 });
            const baseMesh = new THREE.Mesh(baseGeo, baseMat);
            baseMesh.position.y = 0.5;
            baseMesh.castShadow = true;
            baseMesh.receiveShadow = true;
            baseMesh.userData = { seatId: seatId };
            group.add(baseMesh);

            const backGeo = new THREE.BoxGeometry(1.7, 1.7, 0.35);
            const backMat = new THREE.MeshStandardMaterial({ roughness: 0.55 });
            const backMesh = new THREE.Mesh(backGeo, backMat);
            backMesh.position.set(0, 1.3, 0.55);
            backMesh.rotation.x = 0.08;
            backMesh.castShadow = true;
            backMesh.userData = { seatId: seatId };
            group.add(backMesh);

            const armGeo = new THREE.BoxGeometry(0.18, 0.75, 1.4);
            const armMeshL = new THREE.Mesh(armGeo, backMat);
            armMeshL.position.set(-0.9, 0.75, 0.05);
            armMeshL.castShadow = true;
            armMeshL.userData = { seatId: seatId };
            group.add(armMeshL);

            const armMeshR = armMeshL.clone();
            armMeshR.position.x = 0.9;
            armMeshR.userData = { seatId: seatId };
            group.add(armMeshR);

            // الحافة الذهبية الفاخرة الملازمة لتصميم VVIP
            const trimGeo = new THREE.BoxGeometry(0.2, 0.08, 1.42);
            const goldMat = new THREE.MeshStandardMaterial({ color: 0xd97706, metalness: 0.85, roughness: 0.15 });
            const trimL = new THREE.Mesh(trimGeo, goldMat);
            trimL.position.set(-0.9, 1.13, 0.05);
            group.add(trimL);

            const trimR = trimL.clone();
            trimR.position.x = 0.9;
            group.add(trimR);

            // إضافة الأجسام للـ Raycast القابل للضغط
            clickableObjects3D.push(baseMesh, backMesh, armMeshL, armMeshR);

            return group;
        }

        // تحديث الحالات والألوان للـ 3D بناءً على الحجوزات الحقيقية للـ VVIP
        function update3DChairs() {
            if (!scene3D) return;

            const activeEvent = getActiveEvent();
            if (!activeEvent) return;

            chairGroupMap3D.forEach((chairData, seatId) => {
                const group = chairData.group;
                
                // البحث عن الضيف الموافق والمحجوز له هذا الكرسي (الصف كرسي)
                const occupant = activeEvent.guests.find(g => 
                    g.approval === 'approved' && 
                    g.category === 'VVIP' && 
                    g.row === chairData.row && 
                    parseInt(g.seatNum) === chairData.seatNum
                );

                group.children.forEach(child => {
                    if (child.material && child.material !== scene3D.background) {
                        // عدم تعديل الأجزاء الذهبية
                        if (child.material.color && child.material.color.getHex() === 0xd97706 && child.material.metalness > 0.5) {
                            return; 
                        }

                        if (occupant) {
                            child.material.color.setHex(0xb91c1c); // أحمر مخملي فخم للـ VVIP المحجوز
                            child.material.roughness = 0.6;
                        } else {
                            child.material.color.setHex(0x334155); // رمادي غامق أنيق للشاغر
                            child.material.roughness = 0.35;
                        }
                    }
                });

                // إزالة اللوحة العائمة القديمة وإعادة بنائها لمنع تكرار النصوص والأسماء
                const oldSprite = group.getObjectByName("nameplate");
                if (oldSprite) group.remove(oldSprite);

                // إنشاء لوحة الاسم ثلاثية الأبعاد
                const nameSprite = createNameSprite3D(seatId, occupant);
                nameSprite.name = "nameplate";
                nameSprite.position.set(0, 2.5, 0); // تطفو فوق الكرسي
                group.add(nameSprite);
            });
        }

        // توليد لوحات الأسماء ثلاثية الأبعاد بدقة توافقية تامة
        function createNameSprite3D(seatId, guest) {
            const canvas = document.createElement('canvas');
            canvas.width = 380;
            canvas.height = 100;
            const ctx = canvas.getContext('2d');

            // خلفية زجاجية أنيقة متدرجة ومطورة
            const gradient = ctx.createLinearGradient(0, 0, canvas.width, 0);
            if (guest) {
                gradient.addColorStop(0, 'rgba(217, 119, 6, 0.95)');
                gradient.addColorStop(0.5, 'rgba(120, 53, 15, 0.98)');
                gradient.addColorStop(1, 'rgba(217, 119, 6, 0.95)');
            } else {
                gradient.addColorStop(0, 'rgba(51, 65, 85, 0.75)');
                gradient.addColorStop(1, 'rgba(15, 23, 42, 0.85)');
            }

            ctx.fillStyle = gradient;
            
            // استخدام دالة الرسم الآمنة لتجنب انهيار الكود في بعض المتصفحات
            drawRoundRectSafe(ctx, 5, 5, canvas.width - 10, canvas.height - 10, 16);
            ctx.fill();

            // رسم إطار لوحة الاسم
            ctx.lineWidth = 3;
            ctx.strokeStyle = guest ? '#fef3c7' : 'rgba(255,255,255,0.2)';
            ctx.stroke();

            // طباعة المعرف والاسم باللغة العربية والخط المميز
            ctx.fillStyle = guest ? '#fef3c7' : '#94a3b8';
            ctx.font = 'bold 16px Cairo, sans-serif';
            ctx.textAlign = 'center';
            ctx.fillText(`الصف ${seatId}`, canvas.width / 2, 32);

            ctx.fillStyle = '#ffffff';
            ctx.font = 'bold 21px Cairo, sans-serif';
            const displayName = guest ? guest.name : 'مـقـعـد شـاغـر';
            ctx.fillText(displayName, canvas.width / 2, 68);

            // تحويل لوحة الاسم إلى ملمس ثلاثي الأبعاد
            const texture = new THREE.CanvasTexture(canvas);
            const spriteMaterial = new THREE.SpriteMaterial({ 
                map: texture, 
                transparent: true,
                depthTest: true
            });
            
            const sprite = new THREE.Sprite(spriteMaterial);
            sprite.scale.set(4.0, 1.0, 1.0);
            return sprite;
        }

        // النقر على عناصر الكانفاس لتسهيل الحجوزات أو إدارتها
        function onCanvasClick3D(event) {
            if (!scene3D) return;

            const rect = renderer3D.domElement.getBoundingClientRect();
            mouse3D.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
            mouse3D.y = -((event.clientY - rect.top) / rect.height) * 2 + 1;

            raycaster3D.setFromCamera(mouse3D, camera3D);
            const intersects = raycaster3D.intersectObjects(clickableObjects3D);

            if (intersects.length > 0) {
                const hitObj = intersects[0].object;
                if (hitObj.userData && hitObj.userData.seatId) {
                    const seatId = hitObj.userData.seatId; // مثل A-1
                    const parts = seatId.split('-');
                    const rowLetter = parts[0];
                    const seatNum = parseInt(parts[1]);

                    const activeEvent = getActiveEvent();
                    if (!activeEvent) return;

                    // البحث عن الضيف الموافق المسكن في هذا المقعد
                    const occupant = activeEvent.guests.find(g => 
                        g.approval === 'approved' && 
                        g.category === 'VVIP' && 
                        g.row === rowLetter && 
                        parseInt(g.seatNum) === seatNum
                    );
                    
                    if (occupant) {
                        showGuestModal(occupant);
                    } else {
                        // إذا كان المقعد شاغراً، نتيح التسكين السريع للمراسم
                        selectSeatInProtocolForm('VVIP', rowLetter, seatNum);
                    }
                }
            }
        }

        // حلقة التحديث المستمرة للـ 3D
        function animate3D() {
            if (activeViewTab !== '3d') return; // توفير طاقة وموارد المعالج عند تبديل الواجهة

            requestAnimationFrame(animate3D);

            if (controls3D) controls3D.update();

            if (renderer3D && scene3D && camera3D) {
                renderer3D.render(scene3D, camera3D);
            }
        }

        // معالجة تغيير حجم المتصفح لضمان عدم حدوث تمدد مشوه للمقصورة 3D
        function onWindowResize3D() {
            if (!scene3D) return;
            const container = document.getElementById('canvas-3d-holder');
            const width = container.clientWidth || window.innerWidth * 0.55;
            const height = container.clientHeight || 550;

            camera3D.aspect = width / height;
            camera3D.updateProjectionMatrix();
            renderer3D.setSize(width, height);
        }

        // إعادة الكاميرا إلى نقطة البداية المثالية للـ 3D
        function reset3DCamera() {
            if (!camera3D) return;
            camera3D.position.set(0, 19, 29);
            if (controls3D) {
                controls3D.target.set(0, 2, 0);
                controls3D.update();
            }
        }

        function dismiss3DLoading() {
            document.getElementById('canvas3d-loading').classList.add('hidden');
        }

        // توليد أحرف الصفوف ديناميكياً
        function generateAlphabetRows(count, startChar) {
            const startCode = startChar.charCodeAt(0);
            const rows = [];
            for (let i = 0; i < count; i++) {
                rows.push(String.fromCharCode(startCode + i));
            }
            return rows;
        }
